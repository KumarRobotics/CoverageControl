176 def run_data_generation(self):
177 num_non_converged_env = 0
178
179 while self.dataset_count < self.num_dataset:
180 self.env = CoverageSystem(
181 self.env_params, self.env_params.pNumFeatures, self.num_robots
182 )
183 self.alg = CoverageAlgorithm(self.env_params, self.num_robots, self.env)
184 self.env_count += 1
185 print("Environment: " + str(self.env_count))
186 num_steps = 0
187 is_converged = False
188
189 while (
190 num_steps < self.env_params.pEpisodeSteps
191 and not is_converged
192 and self.dataset_count < self.num_dataset
193 ):
194 if num_steps % self.every_num_step == 0:
195 is_converged = self.step_with_save()
196 else:
197 is_converged = self.step_without_save()
198 num_steps += 1
199
200 if num_steps == self.env_params.pEpisodeSteps:
201 num_non_converged_env += 1
202 print("Non-converged environment: " + str(num_non_converged_env))
203
204 num_converged_data = math.ceil(
205 self.converged_data_ratio * num_steps / self.every_num_step
206 )
207 converged_data_count = 0
208
209 while (
210 converged_data_count < num_converged_data
211 and self.dataset_count < self.num_dataset
212 ):
213 self.step_with_save()
214 converged_data_count += 1

216 def step_with_save(self):
217 self.alg.ComputeActions()
218 converged = self.alg.IsConverged()
219 actions = self.alg.GetActions()
220 count = self.dataset_count
221 self.actions[count] = CoverageEnvUtils.to_tensor(actions)
222 self.robot_positions[count] = CoverageEnvUtils.get_robot_positions(self.env)
223 self.coverage_features[count] = CoverageEnvUtils.get_voronoi_features(self.env)
224 self.raw_local_maps[self.trigger_count] = CoverageEnvUtils.get_raw_local_maps(
225 self.env, self.env_params
226 )
227 self.raw_obstacle_maps[self.trigger_count] = (
228 CoverageEnvUtils.get_raw_obstacle_maps(self.env, self.env_params)
229 )
230 self.comm_maps[count] = CoverageEnvUtils.get_communication_maps(
231 self.env, self.env_params, self.cnn_map_size
232 )
233 self.edge_weights[count] = CoverageEnvUtils.get_weights(
234 self.env, self.env_params
235 )
236 self.dataset_count += 1
237
238 if self.dataset_count % 100 == 0:
239 print(f"Dataset: {self.dataset_count}/{self.num_dataset}")
240
241 self.trigger_count += 1
242
243 if self.trigger_count == self.trigger_size:
244 self.trigger_post_processing()
245 self.trigger_count = 0
246
247 error_flag = self.env.StepActions(actions)
248
249 return converged or error_flag

251 def trigger_post_processing(self):
252 if self.trigger_start_idx > self.num_dataset - 1:
253 return
254 trigger_end_idx = min(
255 self.num_dataset, self.trigger_start_idx + self.trigger_size
256 )
257 raw_local_maps = self.raw_local_maps[
258 0 : trigger_end_idx - self.trigger_start_idx
259 ]
260 raw_local_maps = raw_local_maps.to(self.device)
261 resized_local_maps = CoverageEnvUtils.resize_maps(
262 raw_local_maps, self.cnn_map_size
263 )
264 self.local_maps[self.trigger_start_idx : trigger_end_idx] = (
265 resized_local_maps.view(
266 -1, self.num_robots, self.cnn_map_size, self.cnn_map_size
267 )
268 .cpu()
269 .clone()
270 )
271
272 raw_obstacle_maps = self.raw_obstacle_maps[
273 0 : trigger_end_idx - self.trigger_start_idx
274 ]
275 raw_obstacle_maps = raw_obstacle_maps.to(self.device)
276 resized_obstacle_maps = CoverageEnvUtils.resize_maps(
277 raw_obstacle_maps, self.cnn_map_size
278 )
279 self.obstacle_maps[self.trigger_start_idx : trigger_end_idx] = (
280 resized_obstacle_maps.view(
281 -1, self.num_robots, self.cnn_map_size, self.cnn_map_size
282 )
283 .cpu()
284 .clone()
285 )
286
287 self.trigger_start_idx = trigger_end_idx

289 def normalize_tensor(self, tensor):
290 tensor_mean = tensor.mean(dim=[0, 1])
291 tensor_std = tensor.std(dim=[0, 1])
292 tensor = (tensor - tensor_mean) / tensor_std
293
294 return tensor, tensor_mean, tensor_std

296 def normalize_communication_maps(self):
297 min_val = self.comm_maps.min()
298 max_val = self.comm_maps.max()
299 range_val = max_val - min_val
300 self.comm_maps = (self.comm_maps - min_val) / range_val
301 print("Communication map min: " + str(min_val))
302 print("Communication map max: " + str(max_val))
303
304 return min_val, range_val
