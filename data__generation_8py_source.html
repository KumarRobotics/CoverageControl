154 def run_data_generation(self):
155 num_non_converged_env = 0
156
157 while self.dataset_count < self.num_dataset:
158 self.env = CoverageSystem(
159 self.env_params, self.env_params.pNumFeatures, self.num_robots
160 )
161 self.alg = CoverageAlgorithm(self.env_params, self.num_robots, self.env)
162 self.env_count += 1
163 print("Environment: " + str(self.env_count))
164 num_steps = 0
165 is_converged = False
166
167 while (
168 num_steps < self.env_params.pEpisodeSteps
169 and not is_converged
170 and self.dataset_count < self.num_dataset
171 ):
172 if num_steps % self.every_num_step == 0:
173 is_converged = self.step_with_save()
174 else:
175 is_converged = self.step_without_save()
176 num_steps += 1
177
178 if num_steps == self.env_params.pEpisodeSteps:
179 num_non_converged_env += 1
180 print("Non-converged environment: " + str(num_non_converged_env))
181
182 num_converged_data = math.ceil(
183 self.converged_data_ratio * num_steps / self.every_num_step
184 )
185 converged_data_count = 0
186
187 while (
188 converged_data_count < num_converged_data
189 and self.dataset_count < self.num_dataset
190 ):
191 self.step_with_save()
192 converged_data_count += 1

194 def step_with_save(self):
195 self.alg.ComputeActions()
196 converged = self.alg.IsConverged()
197 actions = self.alg.GetActions()
198 count = self.dataset_count
199 self.actions[count] = CoverageEnvUtils.to_tensor(actions)
200 self.robot_positions[count] = CoverageEnvUtils.get_robot_positions(self.env)
201 self.coverage_features[count] = CoverageEnvUtils.get_voronoi_features(self.env)
202 self.raw_local_maps[self.trigger_count] = CoverageEnvUtils.get_raw_local_maps(
203 self.env, self.env_params
204 )
205 self.raw_obstacle_maps[self.trigger_count] = (
206 CoverageEnvUtils.get_raw_obstacle_maps(self.env, self.env_params)
207 )
208 self.comm_maps[count] = CoverageEnvUtils.get_communication_maps(
209 self.env, self.env_params, self.cnn_map_size
210 )
211 self.edge_weights[count] = CoverageEnvUtils.get_weights(
212 self.env, self.env_params
213 )
214 self.dataset_count += 1
215
216 if self.dataset_count % 100 == 0:
217 print(f"Dataset: {self.dataset_count}/{self.num_dataset}")
218
219 self.trigger_count += 1
220
221 if self.trigger_count == self.trigger_size:
222 self.trigger_post_processing()
223 self.trigger_count = 0
224
225 error_flag = self.env.StepActions(actions)
226
227 return converged or error_flag

229 def trigger_post_processing(self):
230 if self.trigger_start_idx > self.num_dataset - 1:
231 return
232 trigger_end_idx = min(
233 self.num_dataset, self.trigger_start_idx + self.trigger_size
234 )
235 raw_local_maps = self.raw_local_maps[
236 0 : trigger_end_idx - self.trigger_start_idx
237 ]
238 raw_local_maps = raw_local_maps.to(self.device)
239 resized_local_maps = CoverageEnvUtils.resize_maps(
240 raw_local_maps, self.cnn_map_size
241 )
242 self.local_maps[self.trigger_start_idx : trigger_end_idx] = (
243 resized_local_maps.view(
244 -1, self.num_robots, self.cnn_map_size, self.cnn_map_size
245 )
246 .cpu()
247 .clone()
248 )
249
250 raw_obstacle_maps = self.raw_obstacle_maps[
251 0 : trigger_end_idx - self.trigger_start_idx
252 ]
253 raw_obstacle_maps = raw_obstacle_maps.to(self.device)
254 resized_obstacle_maps = CoverageEnvUtils.resize_maps(
255 raw_obstacle_maps, self.cnn_map_size
256 )
257 self.obstacle_maps[self.trigger_start_idx : trigger_end_idx] = (

259 -1, self.num_robots, self.cnn_map_size, self.cnn_map_size

261 .cpu()
262 .clone()
263 )

265 self.trigger_start_idx = trigger_end_idx

267 def normalize_tensor(self, tensor):
268 tensor_mean = tensor.mean(dim=[0, 1])
269 tensor_std = tensor.std(dim=[0, 1])
270 tensor = (tensor - tensor_mean) / tensor_std
271
272 return tensor, tensor_mean, tensor_std

274 def normalize_communication_maps(self):
275 min_val = self.comm_maps.min()
276 max_val = self.comm_maps.max()
277 range_val = max_val - min_val
278 self.comm_maps = (self.comm_maps - min_val) / range_val
279 print("Communication map min: " + str(min_val))
280 print("Communication map max: " + str(max_val))
