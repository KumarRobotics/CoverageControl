58__device__ float2 TransformPoint(BND_Cuda const *device_dists, int i,
59 float2 const &in_point) {
60 float2 pt;
61 auto bnd = device_dists[i];
62 if (bnd.rho == 0) {
63 pt.x = (in_point.x - bnd.mean_x) / bnd.sigma_x;
64 pt.y = (in_point.y - bnd.mean_y) / bnd.sigma_y;
65 return pt;
66 }
67 pt.x = (in_point.x - bnd.mean_x) / bnd.sigma_x;
68 pt.y = (in_point.y - bnd.mean_y) / bnd.sigma_y;
69 pt.x = (pt.x - bnd.rho * pt.y) / (sqrt(1 - bnd.rho * bnd.rho));
70 return pt;
71}

73__device__ float IntegrateQuarterPlane(BND_Cuda const &bnd,
74 float2 const &in_point) {
75 float2 pt;
76 if (bnd.rho == 0) {
77 pt.x = (in_point.x - bnd.mean_x) / bnd.sigma_x;
78 pt.y = (in_point.y - bnd.mean_y) / bnd.sigma_y;
79 } else {
80 pt.x = (in_point.x - bnd.mean_x) / bnd.sigma_x;
81 pt.y = (in_point.y - bnd.mean_y) / bnd.sigma_y;
82 pt.x = (pt.x - bnd.rho * pt.y) / (sqrt(1 - bnd.rho * bnd.rho));
83 }
84 /* auto transformed_point = TransformPoint(i, in_point); */
85 float sc = bnd.scale;
86 /* return sc; */
87 return sc * erfc(pt.x / sqrt(2.)) * erfc(pt.y / sqrt(2.)) / 4.;
88}

90__device__ float ComputeImportanceBND(BND_Cuda const *device_dists,
91 float2 const &bottom_left,
92 float2 const &top_right,
93 float2 const &mid_pt_cell) {
94 float2 bottom_right = make_float2(top_right.x, bottom_left.y);
95 float2 top_left = make_float2(bottom_left.x, top_right.y);
96
97 float total_importance = 0;
98 for (int i = 0; i < cu_num_dists; ++i) {
99 auto const &bnd = device_dists[i];
100 float2 mid_pt;
101 if (bnd.rho == 0) {
102 mid_pt.x = (mid_pt_cell.x - bnd.mean_x) / bnd.sigma_x;
103 mid_pt.y = (mid_pt_cell.y - bnd.mean_y) / bnd.sigma_y;
104 } else {
105 mid_pt.x = (mid_pt_cell.x - bnd.mean_x) / bnd.sigma_x;
106 mid_pt.y = (mid_pt_cell.y - bnd.mean_y) / bnd.sigma_y;
107 mid_pt.x =
108 (mid_pt.x - bnd.rho * mid_pt.y) / (sqrt(1 - bnd.rho * bnd.rho));
109 }
110 if (mid_pt.x * mid_pt.x + mid_pt.y * mid_pt.y >
111 cu_truncation * cu_truncation) {
112 continue;
113 }
114 total_importance += IntegrateQuarterPlane(bnd, bottom_left);
115 total_importance -= IntegrateQuarterPlane(bnd, bottom_right);
116 total_importance -= IntegrateQuarterPlane(bnd, top_left);
117 total_importance += IntegrateQuarterPlane(bnd, top_right);
118 }
119 return total_importance;
120}

122__device__ float ComputeImportancePoly(Polygons_Cuda const &device_polygons,
123 float2 const &mid_pt_cell) {
124 float total_importance = 0;
125 int start = 0;
126 auto &x = device_polygons.x;
127 auto &y = device_polygons.y;
128 for (int i = 0; i < cu_num_polygons; ++i) {
129 auto const &bounds = device_polygons.bounds[i];
130 if ((mid_pt_cell.x < bounds.xmin) || (mid_pt_cell.x > bounds.xmax) ||
131 (mid_pt_cell.y < bounds.ymin) || (mid_pt_cell.y > bounds.ymax)) {
132 start += device_polygons.sz[i];
133 continue;
134 }
135 int sz = device_polygons.sz[i];
136 if (IsPointInMonotonePolygon(&x[start], &y[start], sz, mid_pt_cell)) {
137 total_importance = fmaxf(total_importance, device_polygons.imp[i]);
138 }
139 start += sz;
140 }
141 return total_importance;
142}

144__global__ void kernel(BND_Cuda const *device_dists,
145 Polygons_Cuda const device_polygons,
146 float *importance_vec) {
147 int idx = blockIdx.x * blockDim.x + threadIdx.x;
148 int idy = blockIdx.y * blockDim.y + threadIdx.y;
149 int vec_idx = idx * cu_map_size + idy;
150 if (not(idx < cu_map_size and idy < cu_map_size)) {
151 return;
152 }
153 float2 bottom_left = make_float2(idx * cu_resolution, idy * cu_resolution);
154 float2 top_right = make_float2(idx * cu_resolution + cu_resolution,
155 idy * cu_resolution + cu_resolution);
156 float2 mid_pt_cell = make_float2((bottom_left.x + top_right.x) / 2.,
157 (bottom_left.y + top_right.y) / 2.);
158 float poly_importance = ComputeImportancePoly(device_polygons, mid_pt_cell);
159 importance_vec[vec_idx] =
160 ComputeImportanceBND(device_dists, bottom_left, top_right, mid_pt_cell) +
161 poly_importance;
162}
163
164__global__ void normalize(float *importance_vec) {
165 int idx = blockIdx.x * blockDim.x + threadIdx.x;
166 int idy = blockIdx.y * blockDim.y + threadIdx.y;
167 int vec_idx = idx * cu_map_size + idy;
168 if (not(idx < cu_map_size and idy < cu_map_size)) {
169 return;
170 }
171 importance_vec[vec_idx] *= cu_normalization_factor;
172}
173void generate_world_map_cuda(BND_Cuda *host_dists,
174 Polygons_Cuda_Host const &host_polygons,
175 int const num_dists, int const map_size,
176 float const resolution, float const truncation,
177 float const pNorm, float *host_importance_vec,
178 float &normalization_factor) {
179 checkCudaErrors(cudaDeviceSynchronize());
180
181 checkCudaErrors(cudaMemcpyToSymbol(cu_num_dists, &num_dists, sizeof(int)));
182 checkCudaErrors(cudaMemcpyToSymbol(cu_map_size, &map_size, sizeof(int)));
183 checkCudaErrors(
184 cudaMemcpyToSymbol(cu_resolution, &resolution, sizeof(float)));
185 checkCudaErrors(
186 cudaMemcpyToSymbol(cu_truncation, &truncation, sizeof(float)));
187 float f_OneBySqrt2 = static_cast<float>(1. / std::sqrt(2.));
188 checkCudaErrors(
189 cudaMemcpyToSymbol(cu_OneBySqrt2, &f_OneBySqrt2, sizeof(float)));
190
191 BND_Cuda *device_dists;
192 checkCudaErrors(cudaMalloc(&device_dists, num_dists * sizeof(BND_Cuda)));
193 checkCudaErrors(cudaMemcpy(device_dists, host_dists,
194 num_dists * sizeof(BND_Cuda),
195 cudaMemcpyHostToDevice));
196
197 Polygons_Cuda device_polygons;
198 checkCudaErrors(
199 cudaMalloc(&(device_polygons.x), host_polygons.num_pts * sizeof(float)));
200 checkCudaErrors(
201 cudaMalloc(&(device_polygons.y), host_polygons.num_pts * sizeof(float)));
202 checkCudaErrors(cudaMalloc(&(device_polygons.imp),
203 host_polygons.num_polygons * sizeof(float)));
204 checkCudaErrors(cudaMalloc(&(device_polygons.sz),
205 host_polygons.num_polygons * sizeof(int)));
206 checkCudaErrors(cudaMalloc(&(device_polygons.bounds),
207 host_polygons.num_polygons * sizeof(Bounds)));
208
209 checkCudaErrors(cudaMemcpy(device_polygons.x, host_polygons.x.data(),
210 host_polygons.num_pts * sizeof(float),
211 cudaMemcpyHostToDevice));
212 checkCudaErrors(cudaMemcpy(device_polygons.y, host_polygons.y.data(),
213 host_polygons.num_pts * sizeof(float),
214 cudaMemcpyHostToDevice));
215 checkCudaErrors(cudaMemcpy(device_polygons.imp, host_polygons.imp.data(),
216 host_polygons.num_polygons * sizeof(float),
217 cudaMemcpyHostToDevice));
218 checkCudaErrors(cudaMemcpy(device_polygons.sz, host_polygons.sz.data(),
219 host_polygons.num_polygons * sizeof(int),
220 cudaMemcpyHostToDevice));
221 checkCudaErrors(cudaMemcpy(
222 device_polygons.bounds, host_polygons.bounds.data(),
223 host_polygons.num_polygons * sizeof(Bounds), cudaMemcpyHostToDevice));
224 checkCudaErrors(cudaMemcpyToSymbol(cu_polygons_num_pts,
225 &host_polygons.num_pts, sizeof(int)));
226 checkCudaErrors(cudaMemcpyToSymbol(cu_num_polygons,
227 &host_polygons.num_polygons, sizeof(int)));
228
229 float *device_importance_vec;
230 checkCudaErrors(
231 cudaMalloc(&device_importance_vec, map_size * map_size * sizeof(float)));
232
233 /* dim3 dimBlock(1, 1, 1); */
234 /* dim3 dimGrid(1,1,1); */
235
236 dim3 dimBlock(16, 16, 1);
237 dim3 dimGrid(map_size / dimBlock.x, map_size / dimBlock.x, 1);
238
239 kernel<<<dimGrid, dimBlock>>>(device_dists, device_polygons,
240 device_importance_vec);
241
242 cudaDeviceSynchronize();
243
244 thrust::device_ptr<float> d_ptr =
245 thrust::device_pointer_cast(device_importance_vec);
246 float max = *(thrust::max_element(d_ptr, d_ptr + map_size * map_size));
247
248 normalization_factor = pNorm;
249 if (max > 1e-8) {
250 normalization_factor = pNorm / max;
251 }
252 if (normalization_factor > 1e-5) {
253 checkCudaErrors(cudaMemcpyToSymbol(cu_normalization_factor,
254 &normalization_factor, sizeof(float)));
255 normalize<<<dimGrid, dimBlock>>>(device_importance_vec);
256 }
257
258 checkCudaErrors(cudaMemcpy(host_importance_vec, device_importance_vec,
259 map_size * map_size * sizeof(float),
260 cudaMemcpyDeviceToHost));
261
262 checkCudaErrors(cudaFree(device_dists));
263 checkCudaErrors(cudaFree(device_importance_vec));
264 checkCudaErrors(cudaFree(device_polygons.x));
265 checkCudaErrors(cudaFree(device_polygons.y));
266 checkCudaErrors(cudaFree(device_polygons.sz));
267 checkCudaErrors(cudaFree(device_polygons.bounds));
268
269 cudaError_t error = cudaGetLastError();
270 if (error != cudaSuccess) {
271 std::stringstream strstr;
272 strstr << "run_kernel launch failed" << std::endl;
273 throw strstr.str();
274 }
