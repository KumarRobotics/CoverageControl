22"""
23A simple example of using the CoverageControl library
24"""
25
26import coverage_control as cc
27
28# Algorithms available:
29# ClairvoyantCVT
30# CentralizedCVT
31# DecentralizedCVT
32# NearOptimalCVT
33from coverage_control.algorithms import ClairvoyantCVT as CoverageAlgorithm
34
35params = cc.Parameters()
36
37# CoverageSystem handles the environment and robots
38env = cc.CoverageSystem(params)
39
40init_cost = env.GetObjectiveValue()
41print(f"Initial Coverage cost: {init_cost:.2e}")
42
43# Runs the coverage control algorithm
44controller = CoverageAlgorithm(params, env)
45
46for i in range(0, params.pEpisodeSteps):
47 # Compute actions to be taken by the robots
48 controller.ComputeActions()
49 # Get actions from the controller
50 actions = controller.GetActions()
51
52 # Send actions to the environment
53
54 if env.StepActions(actions):
55 print(f"Error in step {i}")
56 break
57
58 if controller.IsConverged():
59 print(f"Converged in step {i}")
60 break
61
62# print some metrics
63current_cost = env.GetObjectiveValue()
64print(f"Improvement %: {100 * (init_cost - current_cost)/init_cost:.2f}")
