37CoverageSystem::CoverageSystem(Parameters const &params,
38 int const num_features, int const num_robots)
39 : params_{params}, world_idf_{WorldIDF(params_)} {
40 // Generate Bivariate Normal Distribution from random numbers
41 std::srand(
42 std::time(nullptr)); // use current time as seed for random generator
43 gen_ = std::mt19937(
44 rd_()); // Standard mersenne_twister_engine seeded with rd_()
45 distrib_pts_ = std::uniform_real_distribution<>(
46 kLargeEps, params_.pWorldMapSize * params_.pResolution - kLargeEps);
47 std::uniform_real_distribution<> distrib_var(params_.pMinSigma,
48 params_.pMaxSigma);
49 std::uniform_real_distribution<> distrib_peak(params_.pMinPeak,
50 params_.pMaxPeak);
51 for (int i = 0; i < num_features; ++i) {
52 Point2 mean(distrib_pts_(gen_), distrib_pts_(gen_));
53 double sigma(distrib_var(gen_));
54 double scale(distrib_peak(gen_));
55 BivariateNormalDistribution dist(mean, sigma, scale);
56 world_idf_.AddNormalDistribution(dist);
57 }
58
59 world_idf_.GenerateMap();
60 normalization_factor_ = world_idf_.GetNormalizationFactor();
61
62 std::uniform_real_distribution<> robot_pos_dist(
63 kLargeEps, params_.pRobotInitDist - kLargeEps);
64 robots_.reserve(num_robots);
65 for (int i = 0; i < num_robots; ++i) {
66 Point2 start_pos(robot_pos_dist(gen_), robot_pos_dist(gen_));
67 robots_.push_back(RobotModel(params_, start_pos, world_idf_));
68 }
69 InitSetup();
70}

130std::pair<MapType, MapType> const &CoverageSystem::GetCommunicationMap(
131 size_t const id, size_t map_size) {
132 communication_maps_[id] = std::make_pair(MapType::Zero(map_size, map_size),
133 MapType::Zero(map_size, map_size));
134 PointVector robot_neighbors_pos = GetRelativePositonsNeighbors(id);
135 double center = map_size / 2. - params_.pResolution / 2.;
136 Point2 center_point(center, center);
137 for (Point2 const &relative_pos : robot_neighbors_pos) {
138 Point2 scaled_indices_val =
139 relative_pos * map_size /
140 (params_.pCommunicationRange * params_.pResolution * 2.) +
141 center_point;
142 int scaled_indices_x = scaled_indices_val[0];
143 int scaled_indices_y = scaled_indices_val[1];
144 Point2 normalized_relative_pos = relative_pos / params_.pCommunicationRange;
145
146 communication_maps_[id].first(scaled_indices_x, scaled_indices_y) +=
147 normalized_relative_pos[0];
148 communication_maps_[id].second(scaled_indices_x, scaled_indices_y) +=
149 normalized_relative_pos[1];
150 }
151 return communication_maps_[id];
152}

154void CoverageSystem::InitSetup() {
155 num_robots_ = robots_.size();
156 robot_positions_history_.resize(num_robots_);
157
158 voronoi_cells_.resize(num_robots_);
159 communication_maps_.resize(num_robots_);
160
161 robot_global_positions_.resize(num_robots_);
162 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
163 robot_global_positions_[iRobot] =
164 robots_[iRobot].GetGlobalCurrentPosition();
165 }
166 system_map_ =
167 MapType::Constant(params_.pWorldMapSize, params_.pWorldMapSize, 0);
168 exploration_map_ =
169 MapType::Constant(params_.pWorldMapSize, params_.pWorldMapSize, 1);
170 explored_idf_map_ =
171 MapType::Constant(params_.pWorldMapSize, params_.pWorldMapSize, 0);
172 total_idf_weight_ = GetWorldIDF().sum();
173 adjacency_matrix_.resize(num_robots_);
174 relative_positions_neighbors_.resize(num_robots_);
175 neighbor_ids_.resize(num_robots_);
176 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
177 adjacency_matrix_[iRobot].resize(num_robots_, 0);
178 relative_positions_neighbors_[iRobot].reserve(num_robots_);
179 neighbor_ids_[iRobot].reserve(num_robots_);
180 }
181 PostStepCommands();
182}

184void CoverageSystem::PostStepCommands(size_t robot_id) {
185 robot_global_positions_[robot_id] =
186 robots_[robot_id].GetGlobalCurrentPosition();
187 for (size_t jRobot = 0; jRobot < robot_id; ++jRobot) {
188 if (robot_id == jRobot) {
189 adjacency_matrix_[robot_id][jRobot] = 0;
190 continue;
191 }
192 Point2 relative_pos =
193 robot_global_positions_[jRobot] - robot_global_positions_[robot_id];
194 if (relative_pos.norm() < params_.pCommunicationRange) {
195 adjacency_matrix_[robot_id][jRobot] = 1;
196 adjacency_matrix_[jRobot][robot_id] = 1;
197 } else {
198 adjacency_matrix_[robot_id][jRobot] = 0;
199 adjacency_matrix_[jRobot][robot_id] = 0;
200 }
201 }
202 UpdateNeighbors();
203
204 if (params_.pUpdateSystemMap) {
205 MapUtils::MapBounds index, offset;
206 MapUtils::ComputeOffsets(
207 params_.pResolution, robot_global_positions_[robot_id],
208 params_.pSensorSize, params_.pWorldMapSize, index, offset);
209 explored_idf_map_.block(index.left + offset.left,
210 index.bottom + offset.bottom, offset.width,
211 offset.height) =
212 GetRobotSensorView(robot_id).block(offset.left, offset.bottom,
213 offset.width, offset.height);
214 exploration_map_.block(
215 index.left + offset.left, index.bottom + offset.bottom, offset.width,
216 offset.height) = MapType::Zero(offset.width, offset.height);
217 system_map_ = explored_idf_map_ - exploration_map_;
218 }
219 auto &history = robot_positions_history_[robot_id];
220 if (history.size() > 0 and
221 history.size() == size_t(params_.pRobotPosHistorySize)) {
222 history.pop_front();
223 } else {
224 history.push_back(robot_global_positions_[robot_id]);
225 }
226}

228void CoverageSystem::PostStepCommands() {
229 UpdateRobotPositions();
230 ComputeAdjacencyMatrix();
231 if (params_.pUpdateSystemMap) {
232 UpdateSystemMap();
233 }
234 UpdateNeighbors();
235 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
236 auto &history = robot_positions_history_[iRobot];
237 if (history.size() > 0 and
238 history.size() == size_t(params_.pRobotPosHistorySize)) {
239 history.pop_front();
240 } else {
241 history.push_back(robot_global_positions_[iRobot]);
242 }
243 }

246void CoverageSystem::ComputeAdjacencyMatrix() {
247 /* #pragma omp parallel for num_threads(num_robots_) */
248 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
249 for (size_t jRobot = 0; jRobot < iRobot; ++jRobot) {
250 if (iRobot == jRobot) {
251 adjacency_matrix_[iRobot][jRobot] = 0;
252 continue;
253 }
254 Point2 relative_pos =
255 robot_global_positions_[jRobot] - robot_global_positions_[iRobot];
256 if (relative_pos.norm() < params_.pCommunicationRange) {
257 adjacency_matrix_[iRobot][jRobot] = 1;
258 adjacency_matrix_[jRobot][iRobot] = 1;
259 } else {
260 adjacency_matrix_[iRobot][jRobot] = 0;
261 adjacency_matrix_[jRobot][iRobot] = 0;
262 }
263 }
264 }

267void CoverageSystem::UpdateNeighbors() {
268 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
269 relative_positions_neighbors_[iRobot].clear();
270 neighbor_ids_[iRobot].clear();
271 for (size_t jRobot = 0; jRobot < num_robots_; ++jRobot) {
272 if (iRobot == jRobot) {
273 continue;
274 }
275 if (adjacency_matrix_[iRobot][jRobot] == 0) {
276 continue;
277 }
278 Point2 relative_pos =
279 robot_global_positions_[jRobot] - robot_global_positions_[iRobot];
280 relative_positions_neighbors_[iRobot].push_back(relative_pos);
281 neighbor_ids_[iRobot].push_back(jRobot);
282 }
283 }

286bool CoverageSystem::StepRobotToGoal(int const robot_id, Point2 const &goal,
287 double const speed_factor) {
288 Point2 curr_pos = robots_[robot_id].GetGlobalCurrentPosition();
289 Point2 diff = goal - curr_pos;
290 double dist = diff.norm();
291 double speed = speed_factor * dist / params_.pTimeStep;
292 if (speed <= kLargeEps) {
293 return 0;
294 }
295 speed = std::min(params_.pMaxRobotSpeed, speed);
296 Point2 direction(diff);
297 direction.normalize();
298 if (robots_[robot_id].StepControl(direction, speed)) {
299 std::cerr << "Control incorrect\n";
300 return 1;
301 }
302 PostStepCommands();
303 return 0;
304}

306bool CoverageSystem::StepRobotsToGoals(PointVector const &goals,
307 PointVector &actions) {
308 bool cont_flag = false;
309 UpdateRobotPositions();
310 /* #pragma omp parallel for num_threads(num_robots_) */
311 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
312 actions[iRobot] = Point2(0, 0);
313 Point2 diff = goals[iRobot] - robot_global_positions_[iRobot];
314 double dist = diff.norm();
315 double speed = dist / params_.pTimeStep;
316 if (speed <= kLargeEps) {
317 continue;
318 }
319 speed = std::min(params_.pMaxRobotSpeed, speed);
320 Point2 direction(diff);
321 direction.normalize();
322 actions[iRobot] = speed * direction;
323 if (StepControl(iRobot, direction, speed)) {
324 std::cerr << "Control incorrect\n";
325 }
326 cont_flag = true;
327 }
328 PostStepCommands();
329 return cont_flag;
330}

332Point2 CoverageSystem::AddNoise(Point2 const pt) const {
333 Point2 noisy_pt;
334 noisy_pt[0] = pt[0];
335 noisy_pt[1] = pt[1];
336 auto noise_sigma = params_.pPositionsNoiseSigma;
337 { // Wrap noise generation in a mutex to avoid issues with random number
338 // generation
339 /* std::lock_guard<std::mutex> lock(mutex_); */
340 std::normal_distribution pos_noise{0.0, noise_sigma};
341 noisy_pt += Point2(pos_noise(gen_), pos_noise(gen_));
342 }
343
344 /* std::normal_distribution pos_noise{0.0, noise_sigma}; */
345 /* noisy_pt += Point2(pos_noise(gen_), pos_noise(gen_)); */
346 if (noisy_pt[0] < kLargeEps) {
347 noisy_pt[0] = kLargeEps;
348 }
349 if (noisy_pt[1] < kLargeEps) {
350 noisy_pt[1] = kLargeEps;
351 }
352 if (noisy_pt[0] > params_.pWorldMapSize - kLargeEps) {
353 noisy_pt[0] = params_.pWorldMapSize - kLargeEps;
354 }
355 if (noisy_pt[1] > params_.pWorldMapSize - kLargeEps) {
356 noisy_pt[1] = params_.pWorldMapSize - kLargeEps;
357 }
358 return noisy_pt;
359}

443void CoverageSystem::RecordPlotData(std::vector<int> const &robot_status,
444 std::string const &map_name) {
445 PlotterData data;
446 if (map_name == "world") {
447 data.map = GetWorldIDF();
448 } else {
449 data.map = system_map_;
450 }
451 data.positions = robot_global_positions_;
452 data.positions_history = robot_positions_history_;
453 data.robot_status = robot_status;
454 ComputeVoronoiCells();
455 std::vector<std::list<Point2>> voronoi;
456 auto voronoi_cells = voronoi_.GetVoronoiCells();
457 for (size_t i = 0; i < num_robots_; ++i) {
458 std::list<Point2> cell_points;

613PointVector CoverageSystem::GetRelativePositonsNeighbors(
614 size_t const robot_id) {
615 if (params_.pAddNoisePositions) {
616 PointVector noisy_positions = GetRobotPositions();
617 for (Point2 &pt : noisy_positions) {
618 pt = AddNoise(pt);
619 }
620 PointVector relative_positions;
621 for (size_t i = 0; i < num_robots_; ++i) {
622 if (i == robot_id) {
623 continue;
624 }
625 if ((noisy_positions[i] - noisy_positions[robot_id]).norm() <
626 params_.pCommunicationRange) {
627 relative_positions.push_back(noisy_positions[i] -
628 noisy_positions[robot_id]);
629 }
630 }
631 return relative_positions;
632 }
633 return relative_positions_neighbors_[robot_id];
634}

636std::vector<double> CoverageSystem::GetLocalVoronoiFeatures(
637 int const robot_id) {
638 auto const &pos = robot_global_positions_[robot_id];
639 MapUtils::MapBounds index, offset;
640 MapUtils::ComputeOffsets(params_.pResolution, pos, params_.pLocalMapSize,
641 params_.pWorldMapSize, index, offset);
642 auto robot_map = robots_[robot_id].GetRobotMap();
643 auto trimmed_local_map =
644 robot_map.block(index.left + offset.left, index.bottom + offset.bottom,
645 offset.width, offset.height);
646 Point2 map_size(offset.width, offset.height);
647
648 Point2 map_translation((index.left + offset.left) * params_.pResolution,
649 (index.bottom + offset.bottom) * params_.pResolution);
650
651 auto robot_neighbors_pos = GetRobotsInCommunication(robot_id);
652 PointVector robot_positions(robot_neighbors_pos.size() + 1);
653
654 robot_positions[0] = pos - map_translation;
655 int count = 1;
656 for (auto const &neighbor_pos : robot_neighbors_pos) {
657 robot_positions[count] = neighbor_pos - map_translation;
658 ++count;
659 }
660 Voronoi voronoi(robot_positions, trimmed_local_map, map_size,
661 params_.pResolution, true, 0);
662 auto vcell = voronoi.GetVoronoiCell();
663 return vcell.GetFeatureVector();
664}
