108 void UpdateSystemMap() {
109 // This is not necessarily thread safe. Do NOT parallelize this for loop
110 for (size_t i = 0; i < num_robots_; ++i) {
111 MapUtils::MapBounds index, offset;
112 MapUtils::ComputeOffsets(params_.pResolution, robot_global_positions_[i],
113 params_.pSensorSize, params_.pWorldMapSize,
114 index, offset);
115 explored_idf_map_.block(index.left + offset.left,
116 index.bottom + offset.bottom, offset.width,
117 offset.height) =
118 GetRobotSensorView(i).block(offset.left, offset.bottom, offset.width,
119 offset.height);
120 exploration_map_.block(
121 index.left + offset.left, index.bottom + offset.bottom, offset.width,
122 offset.height) = MapType::Zero(offset.width, offset.height);
123 }
124 system_map_ = explored_idf_map_ - exploration_map_;
125 /* exploration_ratio_ = 1.0 -
126 * (double)(exploration_map_.sum())/(params_.pWorldMapSize *
127 * params_.pWorldMapSize); */
128 /* weighted_exploration_ratio_ =
129 * (double)(explored_idf_map_.sum())/(total_idf_weight_); */
130 /* std::cout << "Exploration: " << exploration_ratio_ << " Weighted: " <<
131 * weighted_exploration_ratio_ << std::endl; */
132 /* std::cout << "Diff: " << (exploration_map_.count() -
133 * exploration_map_.sum()) << std::endl; */
134 }

333 bool CheckOscillation(size_t const robot_id) const {
334 if (robot_positions_history_[robot_id].size() < 2) {
335 return false;
336 }
337 auto const &history = robot_positions_history_[robot_id];
338 Point2 const last_pos = history.back();
339 auto it_end = std::next(history.crbegin(),
340 std::max(6, static_cast<int>(history.size()) - 1));
341 bool flag = false;
342 int count = 0;
343 std::for_each(history.crbegin(), it_end,
344 [last_pos, &count](Point2 const &pt) {
345 if ((pt - last_pos).norm() < kLargeEps) {
346 ++count;
347 }
348 });
349 if (count > 2) {
350 flag = true;
351 }
352 return flag;
353 }

536 auto GetRobotsInCommunication(size_t const id) const {
537 CheckRobotID(id);
538 PointVector robot_neighbors_pos;
539 for (size_t i = 0; i < num_robots_; ++i) {
540 if (id == i) {
541 continue;
542 }
543 Point2 relative_pos =
544 robot_global_positions_[i] - robot_global_positions_[id];
545 if (relative_pos.norm() < params_.pCommunicationRange) {
546 robot_neighbors_pos.push_back(relative_pos);
547 }
548 }
549 std::sort(
550 robot_neighbors_pos.begin(), robot_neighbors_pos.end(),
551 [](Point2 const &a, Point2 const &b) { return a.norm() < b.norm(); });
552 return robot_neighbors_pos;
553 }
