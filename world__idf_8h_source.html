78 template <class PointType = Point2, typename NumType = double>
79 NumType ComputeImportanceBND(PointType const &bottom_left,
80 PointType const &top_right) const {
81 PointType bottom_right(top_right.x(), bottom_left.y());
82 PointType top_left(bottom_left.x(), top_right.y());
83 NumType importance = 0;
84 PointType mid_point = (bottom_left + top_right) / 2.f;
85 float trun_f = static_cast<float>(params_.pTruncationBND);
86 NumType trun_sqr = trun_f * trun_f;
87 for (auto const &normal_distribution : normal_distributions_) {
88 PointType mid_transformed = normal_distribution.TransformPoint(mid_point);
89 if (mid_transformed[0] * mid_transformed[0] +
90 mid_transformed[1] * mid_transformed[1] >
91 trun_sqr) {
92 continue;
93 }
94 importance += normal_distribution.IntegrateQuarterPlane(bottom_left);
95 importance -= normal_distribution.IntegrateQuarterPlane(bottom_right);
96 importance -= normal_distribution.IntegrateQuarterPlane(top_left);
97 importance += normal_distribution.IntegrateQuarterPlane(top_right);
98 }
99 return importance;
100 }

103 void GenerateMapCuda(float const resolution, float const truncation,
104 int const map_size) {
105 int num_dists = normal_distributions_.size();
106 /* std::cout << "num_dists: " << num_dists << std::endl; */
107
108 /* BND_Cuda *host_dists = (BND_Cuda*) malloc(num_dists * sizeof(BND_Cuda));
109 */
110 BND_Cuda *host_dists = new BND_Cuda[num_dists];
111
112 for (int i = 0; i < num_dists; ++i) {
113 Point2 mean = normal_distributions_[i].GetMean();
114 host_dists[i].mean_x = static_cast<float>(mean.x());
115 host_dists[i].mean_y = static_cast<float>(mean.y());
116 Point2 sigma = normal_distributions_[i].GetSigma();
117 host_dists[i].sigma_x = static_cast<float>(sigma.x());
118 host_dists[i].sigma_y = static_cast<float>(sigma.y());
119 host_dists[i].rho = static_cast<float>(normal_distributions_[i].GetRho());
120 host_dists[i].scale =
121 static_cast<float>(normal_distributions_[i].GetScale());
122 }
123
124 Polygons_Cuda_Host host_polygons;
125 for (auto const &poly_feature : polygon_features_) {
126 std::vector<PointVector> partition_polys;
127 PolygonYMonotonePartition(poly_feature.poly, partition_polys);
128 for (auto const &poly : partition_polys) {
129 Bounds bounds;
130 for (Point2 const &pt : poly) {
131 float x = static_cast<float>(pt.x());
132 float y = static_cast<float>(pt.y());
133 host_polygons.x.push_back(x);
134 host_polygons.y.push_back(y);
135 if (x < bounds.xmin) {
136 bounds.xmin = x;
137 }
138 if (y < bounds.ymin) {
139 bounds.ymin = y;
140 }
141 if (x > bounds.xmax) {
142 bounds.xmax = x;
143 }
144 if (y > bounds.ymax) {
145 bounds.ymax = y;
146 }
147 }
148 host_polygons.imp.push_back(poly_feature.imp);
149 host_polygons.sz.push_back(poly.size());
150 host_polygons.bounds.push_back(bounds);
151 }
152 }
153
154 host_polygons.num_pts = host_polygons.x.size();
155 host_polygons.num_polygons = host_polygons.imp.size();
156
157 float f_norm = 0;
158 generate_world_map_cuda(host_dists, host_polygons, num_dists, map_size,
159 resolution, truncation, params_.pNorm,
160 world_map_.data(), f_norm);
161 normalization_factor_ = static_cast<double>(f_norm);
162 }

179 WorldIDF(Parameters const &params, std::string const &file_name)
180 : WorldIDF(params.pWorldMapSize) {
181 params_ = params;
182 // Load Bivariate Normal Distribution from file
183 params_ = params;
184 std::ifstream file(file_name);
185 if (!file.is_open()) {
186 std::cout << "Error: Could not open file " << file_name << std::endl;
187 exit(1);
188 }
189 std::string type;
190 while (file >> type) {
191 if (type == "BND") {
192 double x, y, sigma_x, sigma_y, rho, scale;
193 file >> x >> y >> sigma_x >> sigma_y >> rho >> scale;
194 AddNormalDistribution(BivariateNormalDistribution(
195 Point2(x, y), Point2(sigma_x, sigma_y), rho, scale));
196 } else if (type == "CircularBND") {
197 double x, y, sigma, scale;
198 file >> x >> y >> sigma >> scale;
199 AddNormalDistribution(
200 BivariateNormalDistribution(Point2(x, y), sigma, scale));
201 } else if (type == "Uniform") {
202 int num_vertices;
203 file >> num_vertices;
204 std::vector<Point2> vertices;
205 for (int i = 0; i < num_vertices; ++i) {
206 double x, y;
207 file >> x >> y;
208 vertices.push_back(Point2(x, y));
209 }
210 double importance;
211 file >> importance;
212 AddUniformDistributionPolygon(PolygonFeature(vertices, importance));
213 } else {
214 std::cout << "Error: Unknown feature type " << type << std::endl;
215 exit(1);
216 }
217 }
218 GenerateMap();
219 }

338 void GenerateMapCPU() {
339 /* std::cout << "Generating map using CPU" << std::endl; */
340 float max_importance = 0;
341 float res = static_cast<float>(params_.pResolution);
342 for (int i = 0; i < params_.pWorldMapSize; ++i) { // Row (x index)
343 float x1 = res * i; // Left x-coordinate of pixel
344 float x2 = x1 + res; // Right x-coordinate of pixel
345 for (int j = 0; j < params_.pWorldMapSize; ++j) { // Column (y index)
346 float y1 = res * j; // Lower y-coordinate of pixel
347 float y2 = y1 + res; // Upper y-coordinate of pixel
348 float importance = ComputeImportanceBND<Point2f, float>(
349 Point2f(x1, y1), Point2f(x2, y2));
350 /* auto importance = ComputeImportanceBND(pt1, pt2); */
351 /* if (std::abs(importance) < kEps) { */
352 /* importance = 0; */
353 /* } */
354 world_map_(i, j) = importance;
355 if (importance > max_importance) {
356 max_importance = importance;
357 }
358 }
359 }
360
361 if (max_importance < kEps) {
362 normalization_factor_ = static_cast<float>(params_.pNorm);
363 } else {
364 normalization_factor_ =
365 static_cast<float>(params_.pNorm) / max_importance;
366 }
367
368 // Normalize the world map
369#pragma omp parallel for
370 for (int i = 0; i < params_.pWorldMapSize; ++i) {
371 for (int j = 0; j < params_.pWorldMapSize; ++j) {
372 world_map_(i, j) *= normalization_factor_;
373 }
374 }
375 }
