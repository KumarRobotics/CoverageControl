77 void GenerateMapCPU() {
78 std::cout << "Generating map using CPU" << std::endl;
79 float max_importance = 0;
80 for (int i = 0; i < params_.pWorldMapSize; ++i) { // Row (x index)
81 float x1 = params_.pResolution * i; // Left x-coordinate of pixel
82 float x2 = x1 + params_.pResolution; // Right x-coordinate of pixel
83 for (int j = 0; j < params_.pWorldMapSize; ++j) { // Column (y index)
84 float y1 = params_.pResolution * j; // Lower y-coordinate of pixel
85 float y2 = y1 + params_.pResolution; // Upper y-coordinate of pixel
86 float importance =
87 ComputeImportanceRectangle(Point2(x1, y1), Point2(x2, y2));
88 if (std::abs(importance) < kEps) {
89 importance = 0;
90 }
91 world_map_(i, j) = importance;
92 if (importance > max_importance) {
93 max_importance = importance;
94 }
95 }
96 }
97
98 if (max_importance < kEps) {
99 normalization_factor_ = params_.pNorm;
100 } else {
101 normalization_factor_ = params_.pNorm / max_importance;
102 }
103
104 // Normalize the world map
105 for (int i = 0; i < params_.pWorldMapSize; ++i) {
106 for (int j = 0; j < params_.pWorldMapSize; ++j) {
107 world_map_(i, j) *= normalization_factor_;
108 }
109 }
110 }

141 Polygons_Cuda_Host host_polygons;
142 for (auto const &poly_feature : polygon_features_) {
143 std::vector<PointVector> partition_polys;
144 PolygonYMonotonePartition(poly_feature.poly, partition_polys);
145 for (auto const &poly : partition_polys) {

148 float x = static_cast<float>(pt.x());
149 float y = static_cast<float>(pt.y());
150 host_polygons.x.push_back(x);

152 if (x < bounds.xmin) {

155 if (y < bounds.ymin) {
156 bounds.ymin = y;

159 bounds.xmax = x;
160 }
161 if (y > bounds.ymax) {
162 bounds.ymax = y;
163 }
164 }
165 host_polygons.imp.push_back(poly_feature.imp);
166 host_polygons.sz.push_back(poly.size());
167 host_polygons.bounds.push_back(bounds);

172 host_polygons.num_polygons = host_polygons.imp.size();

175 generate_world_map_cuda(host_dists, host_polygons, num_dists, map_size,
176 resolution, truncation, params_.pNorm,
177 world_map_.data(), f_norm);
178 normalization_factor_ = static_cast<double>(f_norm);

195 WorldIDF(Parameters const &params, std::string const &file_name)
196 : WorldIDF(params) {
197
198 // Load Bivariate Normal Distribution from file
199 std::ifstream file(file_name);
200 if (!file.is_open()) {
201 std::cout << "Error: Could not open file " << file_name << std::endl;
202 exit(1);
203 }
204 std::string type;
205 while (file >> type) {
206 if (type == "BND") {
207 double x, y, sigma_x, sigma_y, rho, scale;
208 file >> x >> y >> sigma_x >> sigma_y >> rho >> scale;
209 AddNormalDistribution(BivariateNormalDistribution(
210 Point2(x, y), Point2(sigma_x, sigma_y), rho, scale));
211 } else if (type == "CircularBND") {
212 double x, y, sigma, scale;
213 file >> x >> y >> sigma >> scale;
214 AddNormalDistribution(
215 BivariateNormalDistribution(Point2(x, y), sigma, scale));
216 } else if (type == "Uniform") {
217 int num_vertices;
218 file >> num_vertices;
219 std::vector<Point2> vertices;
220 for (int i = 0; i < num_vertices; ++i) {
221 double x, y;
222 file >> x >> y;
223 vertices.push_back(Point2(x, y));
224 }
225 double importance;
226 file >> importance;
227 AddUniformDistributionPolygon(PolygonFeature(vertices, importance));
228 } else {
229 std::cout << "Error: Unknown feature type " << type << std::endl;
230 exit(1);
231 }
232 }

270 float ComputeImportanceRectangle(Point2 const &bottom_left,
271 Point2 const &top_right) const {
272 Point2 bottom_right(top_right.x(), bottom_left.y());
273 Point2 top_left(Point2(bottom_left.x(), top_right.y()));
274 float importance = 0;
275 for (auto const &normal_distribution : normal_distributions_) {
276 Point2 mid_point = (bottom_left + top_right) / 2.;
277 if (normal_distribution.TransformPoint(mid_point).squaredNorm() >
278 params_.pTruncationBND * params_.pTruncationBND) {
279 continue;
280 }
281 importance += normal_distribution.IntegrateQuarterPlane(bottom_left);
282 importance -= normal_distribution.IntegrateQuarterPlane(bottom_right);
283 importance -= normal_distribution.IntegrateQuarterPlane(top_left);
284 importance += normal_distribution.IntegrateQuarterPlane(top_right);
285 }
286 return importance;
287 }

321 inline int WriteDistributions(std::string const &file_name) const {
322 std::ofstream file(file_name);
323 if (!file.is_open()) {
324 std::cerr << "Could not open file: " << file_name << std::endl;
325 return -1;
326 }
327 file << std::setprecision(kMaxPrecision);
328 for (auto const &dist : normal_distributions_) {
329 auto sigma = dist.GetSigma();
330 if (sigma.x() == sigma.y()) {
331 file << "CircularBND" << std::endl;
332 file << dist.GetMean().x() << " " << dist.GetMean().y() << " "
333 << sigma.x() << " " << dist.GetScale() << std::endl;
334 } else {
335 file << "BND" << std::endl;
336 file << dist.GetMean().x() << " " << dist.GetMean().y() << " "
337 << dist.GetSigma().x() << " " << dist.GetSigma().y() << " "
338 << dist.GetRho() << " " << dist.GetScale() << std::endl;
339 }
340 }
341 file.close();
342 return 0;
343 }
