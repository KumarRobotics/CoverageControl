38 bool CheckCudaErrors(cudaError_t result) {
39 if (result != cudaSuccess) {
40 std::cerr << "CUDA error: [" << static_cast<unsigned int>(result) << "] " << cudaGetErrorString(result) << std::endl;
41 return false;
42 }
43 return true;
44 }

55 int CudaUtils::GPUDeviceInit(int dev_id) {
56 int device_count;
57 bool device_count_success = GetDeviceCount(device_count);
58 if (!device_count_success) {
59 return -1;
60 }
61
62 if (dev_id < 0) {
63 dev_id = 0;
64 }
65
66 if (dev_id > device_count - 1) {
67 std::cerr << "Invalid GPU device ID" << std::endl;
68 std::cerr << "Device ID: " << dev_id << " Device Count: " << device_count << std::endl;
69 return -1;
70 }
71
72 int compute_mode = -1, major = 0, minor = 0;
73 CheckCudaErrors(cudaDeviceGetAttribute(&compute_mode, cudaDevAttrComputeMode, dev_id));
74 CheckCudaErrors(cudaDeviceGetAttribute(&major, cudaDevAttrComputeCapabilityMajor, dev_id));
75 CheckCudaErrors(cudaDeviceGetAttribute(&minor, cudaDevAttrComputeCapabilityMinor, dev_id));
76 if (compute_mode == cudaComputeModeProhibited) {
77 std::cerr << "Error: device is running in <Compute Mode Prohibited>, no threads can use cudaSetDevice()." << std::endl;
78 return -1;
79 }
80
81 if (major < 1) {
82 std::cerr << "GPUDeviceInit(): GPU device does not support CUDA." << std::endl;
83 return -1;
84 }
85
86 CheckCudaErrors(cudaSetDevice(dev_id));
87 std::cout << "GPU Device " << dev_id << " has been set" << std::endl;
88 std::cout << "CUDA Device [" << dev_id << "]: \"" << _ConvertSMVer2ArchName(major, minor) << "\"" << std::endl;
89 is_cuda_initialized_ = true;
90 device_id_ = dev_id;
91 return dev_id;
92 }

98 int CudaUtils::FindIntegratedGPU() {
99 int current_device = 0;
100 int device_count = 0;
101 int devices_prohibited = 0;
102
103 CheckCudaErrors(cudaGetDeviceCount(&device_count));
104
105 if (device_count == 0) {
106 /* std::cerr << "No CUDA devices found" << std::endl; */
107 return -1;
108 }
109
110 // Find the integrated GPU which is compute capable
111 while (current_device < device_count) {
112 int compute_mode = -1, integrated = -1;
113 CheckCudaErrors(cudaDeviceGetAttribute(&compute_mode, cudaDevAttrComputeMode, current_device));
114 CheckCudaErrors(cudaDeviceGetAttribute(&integrated, cudaDevAttrIntegrated, current_device));
115 // If GPU is integrated and is not running on Compute Mode prohibited,
116 // then cuda can map to GLES resource
117 if (integrated && (compute_mode != cudaComputeModeProhibited)) {
118 CheckCudaErrors(cudaSetDevice(current_device));
119
120 int major = 0, minor = 0;
121 CheckCudaErrors(cudaDeviceGetAttribute(&major, cudaDevAttrComputeCapabilityMajor, current_device));
122 CheckCudaErrors(cudaDeviceGetAttribute(&minor, cudaDevAttrComputeCapabilityMinor, current_device));
123 std::cout << "GPU Device " << current_device << " has been set" << std::endl;
124 std::cout << "CUDA Device [" << current_device << "]: \"" << _ConvertSMVer2ArchName(major, minor) << "\"" << std::endl;
125 return current_device;
126 } else {
127 devices_prohibited++;
128 }
129
130 current_device++;
131 }
132
133 if (devices_prohibited == device_count) {
134 std::cerr << "CUDA error: No Integrated GPU found that supports CUDA." << std::endl;
135 return -1;
136 }
137
138 return -1;
139
140 }

148 int CudaUtils::GPUGetMaxGflopsDeviceId(std::vector <int> device_list) {
149 int sm_per_multiproc = 0;
150 int max_perf_device = 0;
151 int device_count = 0;
152 int devices_prohibited = 0;
153
154 uint64_t max_compute_perf = 0;
155
156 if(GetDeviceCount(device_count) == false) {
157 return -1;
158 }
159
160 if (device_count == 0) {
161 /* std::cerr << "No CUDA devices found" << std::endl; */
162 return -1;
163 }
164
165 if (device_list.size() == 0) {
166 for (int i = 0; i < device_count; i++) {
167 device_list.push_back(i);
168 }
169 }
170
171 while (device_list.size() > 0) {
172 int current_device = device_list.back();
173 device_list.pop_back();
174 int compute_mode = -1, major = 0, minor = 0;
175 CheckCudaErrors(cudaDeviceGetAttribute(&compute_mode, cudaDevAttrComputeMode, current_device));
176 CheckCudaErrors(cudaDeviceGetAttribute(&major, cudaDevAttrComputeCapabilityMajor, current_device));
177 CheckCudaErrors(cudaDeviceGetAttribute(&minor, cudaDevAttrComputeCapabilityMinor, current_device));
178
179 // If this GPU is not running on Compute Mode prohibited,
180 // then we can add it to the list
181 if (compute_mode != cudaComputeModeProhibited) {
182 if (major == 9999 && minor == 9999) {
183 sm_per_multiproc = 1;
184 } else {
185 sm_per_multiproc =
186 _ConvertSMVer2Cores(major, minor);
187 }
188 int multiProcessorCount = 0, clockRate = 0;
189 CheckCudaErrors(cudaDeviceGetAttribute(&multiProcessorCount, cudaDevAttrMultiProcessorCount, current_device));
190 cudaError_t result = cudaDeviceGetAttribute(&clockRate, cudaDevAttrClockRate, current_device);
191 if (result != cudaSuccess) {
192 // If cudaDevAttrClockRate attribute is not supported we
193 // set clockRate as 1, to consider GPU with most SMs and CUDA Cores.
194 if(result == cudaErrorInvalidValue) {
195 clockRate = 1;
196 }
197 else {
198 fprintf(stderr, "CUDA error at %s:%d code=%d(%s) \n", __FILE__, __LINE__,
199 static_cast<unsigned int>(result), _cudaGetErrorEnum(result));
200 return -1;
201 }
202 }
203 uint64_t compute_perf = (uint64_t)multiProcessorCount * sm_per_multiproc * clockRate;
204
205 if (compute_perf > max_compute_perf) {
206 max_compute_perf = compute_perf;
207 max_perf_device = current_device;
208 }
209 } else {
210 devices_prohibited++;
211 }
212 }
213
214 if (devices_prohibited == device_count) {
215
216 std::cerr << "GPUGetMaxGflopsDeviceId() CUDA error:"
217 << " all devices have compute mode prohibited." << std::endl;
218 return -1;
219 }
220
221 return max_perf_device;
222 }
