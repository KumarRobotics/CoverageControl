44__device__ int Orientation(float2 const &p, float2 const &q, float2 const &r) {
45 float2 qp{q.x - p.x, q.y - p.y};
46 float2 rp{r.x - p.x, r.y - p.y};
47 float cross_prod1 = qp.x * rp.y;
48 float cross_prod2 = qp.y * rp.x;
49 if (cross_prod1 > cross_prod2) return 1;
50 if (cross_prod1 < cross_prod2) return -1;
51 return 0;
52}

58__device__ bool IsPointInMonotonePolygon(float *x, float *y, int sz,
59 float2 const &r) {
60 bool left = false;
61 bool right = false;
62 for (int ct = 0; ct < sz; ++ct) {
63 int ctp1 = ct + 1;
64 if (ctp1 == sz) {
65 ctp1 = 0;
66 }
67 float yct = y[ct];
68 float yctp1 = y[ctp1];
69 /* printf("Edge: %f, %f, %f, %f\n", x[ct], yct, x[ctp1], yctp1); */
70 float2 p, q;
71 if (yct < yctp1) {
72 if (r.y < yct or r.y > yctp1) {
73 continue;
74 }
75 p = float2{x[ct], yct};
76 q = float2{x[ctp1], yctp1};
77 } else {
78 if (r.y < yctp1 or r.y > yct) {
79 continue;
80 }
81 q = float2{x[ct], yct};
82 p = float2{x[ctp1], yctp1};
83 }
84 int orientation = Orientation(p, q, r);
85 /* printf("orientation: %d\n", orientation); */
86 if (orientation == 0) {
87 return true;
88 }
89 if (orientation == 1) {
90 left = true;
91 }
92 if (orientation == -1) {
93 right = true;
94 }
95 if ((left & right)) {
96 return true;
97 }
98 }
99 return false;
100}
