22# @file simple_data_generation.py
23# This file contains the code to generate a dataset for learning
24#
25
26# Uses the following configuration given in a toml file:
27# DataDir = "${CoverageControl_ws}/datasets/lpac" # Absolute location
28# EnvironmentConfig = "${CoverageControl_ws}/datasets/lpac/coverage_control_params.toml" # Absolute location
29#
30# NumDataset = 1000
31#
32# # Number of steps to take before data is stores
33# # This helps in creating a more diverse dataset
34# EveryNumSteps = 5
35#
36# # The robots stop moving once the algorithm has converged
37# # Having some of these converged steps can help in stabilizing robot actions
38# ConvergedDataRatio = 0.02
39#
40# # Resizing of maps and Sparsification of tensors are triggered every TriggerPostProcessing dataset
41# # This should be set based on RAM resources available on the system
42# TriggerPostProcessing = 100
43#
44# CNNMapSize = 32

135 def run_data_generation(self):
136 num_non_converged_env = 0
137 while self.dataset_count < self.num_dataset:
138 self.env = CoverageSystem(self.env_params, self.env_params.pNumFeatures, self.num_robots)
139 self.alg = CoverageAlgorithm(self.env_params, self.num_robots, self.env)
140 self.env_count += 1
141 print('Environment: ' + str(self.env_count))
142 num_steps = 0
143 is_converged = False
144 while num_steps < self.env_params.pEpisodeSteps and not is_converged and self.dataset_count < self.num_dataset:
145 if num_steps % self.every_num_step == 0:
146 is_converged = self.step_with_save()
147 else:
148 is_converged = self.step_without_save()
149 num_steps += 1
150 if num_steps == self.env_params.pEpisodeSteps:
151 num_non_converged_env += 1
152 print('Non-converged environment: ' + str(num_non_converged_env))
153
154 print('Converged in ' + str(num_steps) + ' steps')
155
156 num_converged_data = math.ceil(self.converged_data_ratio * num_steps / self.every_num_step)
157 converged_data_count = 0
158 while converged_data_count < num_converged_data and self.dataset_count < self.num_dataset:
159 self.step_with_save()
160 converged_data_count += 1

162 def step_with_save(self):
163 self.alg.ComputeActions()
164 converged = self.alg.IsConverged()
165 actions = self.alg.GetActions()
166 count = self.dataset_count
167 self.actions[count] = CoverageEnvUtils.to_tensor(actions)
168 self.robot_positions[count] = CoverageEnvUtils.get_robot_positions(self.env)
169 self.coverage_features[count] = CoverageEnvUtils.get_voronoi_features(self.env)
170 self.raw_local_maps[self.trigger_count] = CoverageEnvUtils.get_raw_local_maps(self.env, self.env_params)
171 self.raw_obstacle_maps[self.trigger_count] = CoverageEnvUtils.get_raw_obstacle_maps(self.env, self.env_params)
172 self.comm_maps[count] = CoverageEnvUtils.get_communication_maps(self.env, self.env_params, self.cnn_map_size)
173 self.edge_weights[count] = CoverageEnvUtils.get_weights(self.env, self.env_params)
174 self.dataset_count += 1
175 if self.dataset_count % 100 == 0:
176 print(f'Dataset: {self.dataset_count}/{self.num_dataset}')
177
178 self.trigger_count += 1
179 if self.trigger_count == self.trigger_size:
180 self.trigger_post_processing()
181 self.trigger_count = 0
182
183 if(self.env.StepActions(actions)):
184 return True
185 return converged

187 def trigger_post_processing(self):
188 if self.trigger_start_idx > self.num_dataset -1:
189 return
190 trigger_end_idx = min(self.num_dataset, self.trigger_start_idx + self.trigger_size)
191 raw_local_maps = self.raw_local_maps[0:trigger_end_idx - self.trigger_start_idx]
192 raw_local_maps = raw_local_maps.to(self.device)
193 resized_local_maps = CoverageEnvUtils.resize_maps(raw_local_maps, self.cnn_map_size)
194 self.local_maps[self.trigger_start_idx:trigger_end_idx] = resized_local_maps.view(-1, self.num_robots, self.cnn_map_size, self.cnn_map_size).cpu().clone()
195
196 raw_obstacle_maps = self.raw_obstacle_maps[0:trigger_end_idx - self.trigger_start_idx]
197 raw_obstacle_maps = raw_obstacle_maps.to(self.device)
198 resized_obstacle_maps = CoverageEnvUtils.resize_maps(raw_obstacle_maps, self.cnn_map_size)
199 self.obstacle_maps[self.trigger_start_idx:trigger_end_idx] = resized_obstacle_maps.view(-1, self.num_robots, self.cnn_map_size, self.cnn_map_size).cpu().clone()
200
201 self.trigger_start_idx = trigger_end_idx

203 def normalize_tensor(self, tensor):
204 tensor_mean = tensor.mean(dim=[0, 1])
205 tensor_std = tensor.std(dim=[0, 1])
206 tensor = (tensor - tensor_mean) / tensor_std
207 return tensor, tensor_mean, tensor_std

209 def normalize_communication_maps(self):
210 min_val = self.comm_maps.min()
211 max_val = self.comm_maps.max()
212 range_val = max_val - min_val
213 self.comm_maps = (self.comm_maps - min_val) / range_val
214 print('Communication map min: ' + str(min_val))
215 print('Communication map max: ' + str(max_val))
216 return min_val, range_val
