57inline void ComputeOffsets(double const resolution, Point2 const &pos,
58 int const submap_size, int const map_size,
59 MapBounds &index, MapBounds &offset) {
60 int pos_idx = 0, pos_idy = 0;
61 GetClosestGridCoordinate(resolution, pos, pos_idx, pos_idy);
62 index.left = pos_idx - submap_size / 2;
63 index.right = pos_idx + submap_size / 2;
64 index.bottom = pos_idy - submap_size / 2;
65 index.top = pos_idy + submap_size / 2;
66
67 offset.SetZero();
68 if (index.left < 0) {
69 offset.left = -index.left;
70 }
71 if (index.bottom < 0) {
72 offset.bottom = -index.bottom;
73 }
74
75 if (index.right > map_size) {
76 offset.right = map_size - index.right;
77 }
78 if (index.top > map_size) {
79 offset.top = map_size - index.top;
80 }
81
82 offset.width = index.right + offset.right - (index.left + offset.left);
83 offset.height = index.top + offset.top - (index.bottom + offset.bottom);
84}

86template <typename T = MapType>
87inline void GetSubMap(double const resolution, Point2 const &pos,
88 int const map_size, T const &map, int const submap_size,
89 T &submap) {
90 MapBounds index, offset;
91 ComputeOffsets(resolution, pos, submap_size, map_size, index, offset);
92 submap.block(offset.left, offset.bottom, offset.width, offset.height) =
93 map.block(index.left + offset.left, index.bottom + offset.bottom,
94 offset.width, offset.height);
95}

103 offset.width, offset.height);

120inline int WriteMapSparse(MapType const &map, std::string const &file_name) {
121 std::ofstream file_obj(file_name);
122 if (!file_obj) {
123 std::cerr << "[Error] Could not open " << file_name << " for writing."
124 << std::endl;
125 return 1;
126 }
127 for (int i = 0; i < map.rows(); ++i) {
128 for (int j = 0; j < map.cols(); ++j) {
129 if (map(i, j) >= 0) {
130 file_obj << i << " " << j << " " << map(i, j) << std::endl;
131 }
132 }
133 }
134 file_obj.close();
135 return 0;
136}

138inline int IsPointOutsideBoundary(double const resolution, Point2 const &pos,
139 int const sensor_size, int const boundary) {
140 if (pos.x() <= -sensor_size * resolution / 2.) {
141 return 1;
142 }
143 if (pos.y() <= -sensor_size * resolution / 2.) {
144 return 1;
145 }
146 if (pos.x() >= boundary * resolution + sensor_size * resolution / 2.) {
147 return 1;
148 }
149 if (pos.y() >= boundary * resolution + sensor_size * resolution / 2.) {
150 return 1;
151 }
152 return 0;
153}
