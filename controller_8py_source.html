37class ControllerCVT:
38 """
39 Controller class for CVT based controllers
40 """
41
42 def __init__(self, config: dict, params: Parameters, env: CoverageSystem):
43 """
44 Constructor for the CVT controller
45 Args:
46 config: Configuration dictionary
47 params: Parameters object
48 env: CoverageSystem object
49 """
50 self.name = config["Name"]
51 self.params = params
52 match config["Algorithm"]:
53 case "DecentralizedCVT":
54 self.alg = DecentralizedCVT(params, env)
55 case "ClairvoyantCVT":
56 self.alg = ClairvoyantCVT(params, env)
57 case "CentralizedCVT":
58 self.alg = CentralizedCVT(params, env)
59 case "NearOptimalCVT":
60 self.alg = NearOptimalCVT(params, env)
61 case _:
62 raise ValueError(f"Unknown controller type: {controller_type}")
63
64 def step(self, env: CoverageSystem) -> (float, bool):
65 """
66 Step function for the CVT controller
67
68 Performs three steps:
69 1. Compute actions using the CVT algorithm
70 2. Get the actions from the algorithm
71 3. Step the environment using the actions
72 Args:
73 env: CoverageSystem object
74 Returns:
75 Objective value and convergence flag
76 """
77 self.alg.ComputeActions()
78 actions = self.alg.GetActions()
79 converged = self.alg.IsConverged()
80 error_flag = env.StepActions(actions)
81
82 if error_flag:
83 raise ValueError("Error in step")
84
85 return env.GetObjectiveValue(), converged

93 def __init__(self, config: dict, params: Parameters, env: CoverageSystem):
94 """
95 Constructor for the neural network controller
96 Args:
97 config: Configuration dictionary
98 params: Parameters object
99 env: CoverageSystem object
100 """
101 self.config = config
102 self.params = params
103 self.name = self.config["Name"]
104 self.use_cnn = self.config["UseCNN"]
105 self.use_comm_map = self.config["UseCommMap"]
106 self.cnn_map_size = self.config["CNNMapSize"]
107
108 self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
109 print(f"Using device: {self.device}")
110
111 if "ModelFile" in self.config:
112 self.model_file = IOUtils.sanitize_path(self.config["ModelFile"])
113 self.model = torch.load(self.model_file)
114 else: # Load from ModelStateDict
115 self.learning_params_file = IOUtils.sanitize_path(
116 self.config["LearningParams"]
117 )
118 self.learning_params = IOUtils.load_toml(self.learning_params_file)
119 self.model = cc_nn.LPAC(self.learning_params).to(self.device)
120 self.model.load_model(IOUtils.sanitize_path(self.config["ModelStateDict"]))
121
122 self.actions_mean = self.model.actions_mean.to(self.device)
123 self.actions_std = self.model.actions_std.to(self.device)
124 self.model = self.model.to(self.device)
125 self.model.eval()
126
127 def step(self, env):
128 """
129 step function for the neural network controller
130
131 Performs three steps:
132 1. Get the data from the environment
133 2. Get the actions from the model
134 3. Step the environment using the actions
135 Args:
136 env: CoverageSystem object
137 Returns:
138 Objective value and convergence flag
139 """
140 pyg_data = CoverageEnvUtils.get_torch_geometric_data(
141 env, self.params, True, self.use_comm_map, self.cnn_map_size
142 ).to(self.device)
143 with torch.no_grad():
144 actions = self.model(pyg_data)
145 actions = actions * self.actions_std + self.actions_mean
146 point_vector_actions = PointVector(actions.cpu().numpy())
147 env.StepActions(point_vector_actions)
148
149 return env.GetObjectiveValue(), False
