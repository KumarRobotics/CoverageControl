86 WorldIDF world_idf(params, idf_file);
87 env = std::make_unique<CoverageSystem>(params, world_idf, pos_file);

89 env = std::make_unique<CoverageSystem>(params);

91
92 auto init_objective = env->GetObjectiveValue();
93 std::cout << "Initial objective: " << init_objective << std::endl;
94
95 CoverageAlgorithm algorithm(params, *env);
96 auto goals = algorithm.GetGoals();
97
98 for (int ii = 0; ii < params.pEpisodeSteps; ++ii) {
99 algorithm.ComputeActions();
100 auto actions = algorithm.GetActions();
101 if (env->StepActions(actions)) {
102 std::cout << "Invalid action" << std::endl;
103 break;
104 }
105 if (ii % 100 == 0) {
106 std::cout << "Step: " << ii << std::endl;
107 }
108 if (algorithm.IsConverged()) {
109 break;
110 }
111 }
112 auto final_objective = env->GetObjectiveValue();
113 std::cout << "Improvement %: "
114 << (init_objective - final_objective) / init_objective * 100
115 << std::endl;
116 return 0;
