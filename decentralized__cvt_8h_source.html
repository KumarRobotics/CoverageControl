75 DecentralizedCVT(Parameters const &params, size_t const &num_robots,

89 void ComputeGoals() {
90#pragma omp parallel for num_threads(num_robots_)
91 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
92 auto const &pos = robot_global_positions_[iRobot];
93 MapUtils::MapBounds index, offset;
94 MapUtils::ComputeOffsets(params_.pResolution, pos, params_.pLocalMapSize,
95 params_.pWorldMapSize, index, offset);
96 auto robot_map = env_.GetRobotMap(iRobot);
97 auto robot_local_map = robot_map.block(index.left + offset.left,
98 index.bottom + offset.bottom,
99 offset.width, offset.height);
100 Point2 map_translation(
101 (index.left + offset.left) * params_.pResolution,
102 (index.bottom + offset.bottom) * params_.pResolution);
103
104 PointVector robot_neighbors_pos;
105 for (size_t jRobot = 0; jRobot < num_robots_; ++jRobot) {
106 if (iRobot == jRobot) {
107 continue;
108 }
109 Point2 relative_pos =
110 robot_global_positions_[jRobot] - robot_global_positions_[iRobot];
111 if (relative_pos.norm() < params_.pCommunicationRange) {
112 robot_neighbors_pos.push_back(robot_global_positions_[jRobot]);
113 }
114 }
115 PointVector robot_positions(robot_neighbors_pos.size() + 1);
116
117 robot_positions[0] = robot_global_positions_[iRobot] - map_translation;
118 /* std::cout << "Voronoi: " << robot_positions[0][0] << " " <<
119 * robot_positions[0][1] << std::endl; */
120 int count = 1;
121 for (auto const &pos : robot_neighbors_pos) {
122 robot_positions[count] = pos - map_translation;
123 ++count;
124 }
125 Point2 map_size(offset.width, offset.height);
126 Voronoi voronoi(robot_positions, robot_local_map, map_size,
127 params_.pResolution, true, 0);
128 auto vcell = voronoi.GetVoronoiCell();
129 voronoi_mass_[iRobot] = vcell.mass();
130 goals_[iRobot] = vcell.centroid() + robot_positions[0] + map_translation;
131 if (goals_[iRobot][0] < 0 or goals_[iRobot][0] > params_.pWorldMapSize or
132 goals_[iRobot][1] < 0 or goals_[iRobot][1] > params_.pWorldMapSize) {
133 std::cout << "Goal out of bounds: " << goals_[iRobot][0] << " "
134 << goals_[iRobot][1] << std::endl;
135 std::cout << robot_global_positions_[iRobot][0] << " "
136 << robot_global_positions_[iRobot][1] << std::endl;
137 std::cout << vcell.centroid()[0] << " " << vcell.centroid()[1]
138 << std::endl;
139 std::cout << map_translation[0] << " " << map_translation[1]
140 << std::endl;
141 std::cout << robot_local_map.sum() << std::endl;
142 throw std::runtime_error(
143 "Goal out of bounds: this should not have happened for convex "
144 "environments");
145 }
146 }
147 }

149 bool ComputeActions() {
150 is_converged_ = true;
151 robot_global_positions_ = env_.GetRobotPositions();
152 ComputeGoals();
153 for (size_t iRobot = 0; iRobot < num_robots_; ++iRobot) {
154 actions_[iRobot] = Point2(0, 0);
155 Point2 diff = goals_[iRobot] - robot_global_positions_[iRobot];
156 double dist = diff.norm();
157 /* std::cout << "Robot " << iRobot << " goal: " << goals_[iRobot][0] <<
158 * " " << goals_[iRobot][1] << " " << dist << std::endl; */
159 if (dist < kEps) {
160 continue;
161 }
162 if (env_.CheckOscillation(iRobot)) {
163 continue;
164 }
165 double speed = dist / params_.pTimeStep;
166 /* double speed = 2 * dist * voronoi_mass_[iRobot]; */
167 /* if (dist > 0 and voronoi_mass_[iRobot] == 0) { */
168 /* speed = dist / params_.pTimeStep; */
169 /* } */
170 speed = std::min(params_.pMaxRobotSpeed, speed);
171 Point2 direction(diff);
172 direction.normalize();
173 actions_[iRobot] = speed * direction;
174 is_converged_ = false;
175 return true;
176 }
177 }
178
179 bool IsConverged() const { return is_converged_; }
