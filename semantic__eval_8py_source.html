77 robot_positions = CoverageSystemUtils.GetRobotPositions(env)
78 robot_positions = (robot_positions + params.pWorldMapSize/2) / params.pWorldMapSize

80 data = dl_utils.ToTorchGeometricData(maps, edge_weights, robot_positions)
