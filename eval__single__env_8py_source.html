38 Class to evaluate a single environment with multiple controllers

41 def __init__(self, in_config):
42 self.config = in_config
43 self.eval_dir = IOUtils.sanitize_path(self.config["EvalDir"]) + "/"
44 self.env_dir = IOUtils.sanitize_path(self.config["EnvironmentDataDir"]) + "/"
45
46 if not os.path.exists(self.env_dir):
47 os.makedirs(self.env_dir)
48
49 self.num_controllers = len(self.config["Controllers"])
50 self.controllers_configs = self.config["Controllers"]
51
52 self.env_config_file = IOUtils.sanitize_path(self.config["EnvironmentConfig"])
53 self.env_config = IOUtils.load_toml(self.env_config_file)
54 self.cc_params = cc.Parameters(self.env_config_file)
55
56 self.num_steps = self.config["NumSteps"]
57
58 self.plot_map = self.config["PlotMap"]
59 self.generate_video = self.config["GenerateVideo"]
60
61 self.feature_file = None
62 self.pos_file = None
63
64 file_exists = False
65 if "FeatureFile" in self.config and "RobotPosFile" in self.config:
66 self.feature_file = self.env_dir + self.config["FeatureFile"]
67 self.pos_file = self.env_dir + self.config["RobotPosFile"]
68 print(self.feature_file)
69 print(self.pos_file)
70 if os.path.isfile(self.feature_file):
71 feature_file_exists = True
72 else:
73 print(f"Feature file {self.feature_file} not found")
74 feature_file_exists = False
75 if os.path.isfile(self.pos_file):
76 pos_file_exists = True
77 else:
78 print(f"Position file {self.pos_file} not found")
79 pos_file_exists = False
80
81 file_exists = feature_file_exists and pos_file_exists
82
83 if file_exists is True:
84 self.world_idf = WorldIDF(self.cc_params, self.feature_file)
85 self.env_main = CoverageSystem(
86 self.cc_params, self.world_idf, self.pos_file
87 )
88 else:
89 self.env_main = CoverageSystem(self.cc_params)
90 self.world_idf = self.env_main.GetWorldIDF()
91 if self.feature_file is not None and self.pos_file is not None:
92 self.env_main.WriteEnvironment(self.pos_file, self.feature_file)
93
94 def evaluate(self, save=True):
95 cost_data = np.zeros((self.num_controllers, self.num_steps))
96 robot_init_pos = self.env_main.GetRobotPositions(force_no_noise=True)
97
98 if self.plot_map:
99 map_dir = self.eval_dir + "/plots/"
100 os.makedirs(map_dir, exist_ok=True)
101 self.env_main.PlotInitMap(map_dir, "InitMap")
102
103 for controller_id in range(self.num_controllers):
104 step_count = 0
105 env = CoverageSystem(self.cc_params, self.world_idf, robot_init_pos)
106 controller_name = self.controllers_configs[controller_id]["Name"]
107
108 if self.generate_video:
109 env.RecordPlotData()
110
111 if self.controllers_configs[controller_id]["Type"] == "Learning":
112 Controller = ControllerNN
113 else:
114 Controller = ControllerCVT
115 controller = Controller(
116 self.controllers_configs[controller_id], self.cc_params, env
117 )
118 initial_objective_value = env.GetObjectiveValue()
119 cost_data[controller_id, step_count] = (
120 env.GetObjectiveValue() / initial_objective_value
121 )
122 step_count = step_count + 1
123
124 while step_count < self.num_steps:
125 objective_value, converged = controller.step(env)
126 cost_data[controller_id, step_count] = (
127 objective_value / initial_objective_value
128 )
129
130 if converged and not self.generate_video:
131 cost_data[controller_id, step_count:] = (
132 objective_value / initial_objective_value
133 )
134
135 break
136
137 if self.generate_video:
138 env.RecordPlotData()
139
140 step_count = step_count + 1
141
142 if step_count % 100 == 0:
143 print(
144 f"{controller.name} "
145 f"Step {step_count} "
146 f"Obj Value {cost_data[controller_id, step_count - 1]:.3e}"
147 )
148 print(
149 f"{controller.name} "
150 f"final step {step_count} "
151 f"Obj Value {cost_data[controller_id, step_count - 1]:.3e}"
152 )
153
154 if save is True:
155 controller_dir = self.eval_dir + "/" + controller_name
156
157 if not os.path.exists(controller_dir):
158 os.makedirs(controller_dir)
159
160 controller_data_file = controller_dir + "/" + "eval.csv"
161 np.savetxt(
162 controller_data_file, cost_data[controller_id, :], delimiter=","
163 )
164
165 if self.generate_video:
166 controller_dir = self.eval_dir + "/" + controller_name
167 env.RenderRecordedMap(controller_dir, "video.mp4")
168
169 del controller
170 del env
171
172 return cost_data
