34 std::cout << std::boolalpha;
35 std::cout << "Using config file: " << config_file_ << std::endl;
36 if (not std::filesystem::exists(config_file_)) {
37 std::cerr << "Could not find config file " << config_file_ << std::endl;
38 throw std::runtime_error("Could not open config file");
39 }
40
41 toml::table toml_config;
42 try {
43 toml_config = toml::parse_file(config_file_);
44 } catch (const std::exception& e) {
45 std::cerr << "Error parsing config file: " << e.what() << std::endl;

200 pPositionsNoiseSigma = toml_PositionsNoiseSigma.value();
201 }

204 auto toml_Algorithm = toml_config["Algorithm"];
205
206 if (toml_Algorithm) {
207 auto toml_EpisodeSteps = toml_Algorithm["EpisodeSteps"].value<int>();
208 if (toml_EpisodeSteps) {
209 pEpisodeSteps = toml_EpisodeSteps.value();
210 }

214 auto toml_LloydNumTries =
215 toml_Algorithm["Global-CVT.LloydNumTries"].value<int>();

217 pLloydMaxIterations = toml_LloydMaxIterations.value();

220 pLloydNumTries = toml_LloydNumTries.value();

223 auto toml_NumFrontiers =

225 if (toml_NumFrontiers) {
226 pNumFrontiers = toml_NumFrontiers.value();
227 }
228 }
