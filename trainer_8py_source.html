163 def TrainEpoch(self) -> float:
164 """
165 Train the model in batches
166
167 Returns:
168 training loss
169 """
170 # Initialize the training loss
171 train_loss = 0.0
172
173 # Set the model to training mode
174 self.model.train()
175
176 num_dataset = 0
177 # Train the model in batches
178
179 for batch_idx, (data, target) in enumerate(self.train_loader):
180 # Move the data to the device
181 data, target = data.to(self.device), target.to(self.device)
182
183 if target.dim() == 3:
184 target = target.view(-1, target.shape[-1])
185
186 # Clear the gradients
187 self.optimizer.zero_grad()
188
189 # Forward propagation
190 output = self.model(data)
191
192 # Calculate the loss
193 loss = self.criterion(output, target)
194
195 # Print batch number and loss
196
197 if batch_idx % 10 == 0:
198 print("Batch: {}, Loss: {}".format(batch_idx, loss))
199
200 # Backward propagation
201 loss.backward()
202
203 # Update the parameters
204 self.optimizer.step()
205
206 # Update the training loss
207 # train_loss += loss.item() * data.size(0)
208 # num_dataset += data.size(0)
209 train_loss += loss.item()
210 num_dataset += 1
211
212 # Return the training loss

214 return train_loss / num_dataset
