181 int StepControl(Point2 const &direction, double const &speed) {
182 auto dir = direction;
183 auto sp = speed;
184 Point2 new_pos = local_current_position_;
185 if (sp > params_.pMaxRobotSpeed) {
186 sp = params_.pMaxRobotSpeed;
187 }
188 bool invalid_dir = dir.norm() < kEps and sp >= kEps;
189 if (sp < 0 or invalid_dir) {
190 std::cout << sp << " " << dir.norm() << std::endl;
191 std::cerr << "Speed needs to be non-negative\n";
192 std::cerr << "Zero-vector direction cannot be given in control\n";
193 std::throw_with_nested(
194 std::runtime_error("Speed needs to be non-negative"));
195 return 1;
196 }
197 /* if (dir.normalize() and sp > kEps) { */
198 /* std::cerr << "Zero-vector direction given in control\n"; */
199 /* return 1; */
200 /* } */
201 dir.normalize();
202 if (sp > kEps) {
203 new_pos = local_current_position_ + dir * sp * params_.pTimeStep;
204 }
205 SetRobotPosition(new_pos);
206 return 0;
207 }

210 void SetRobotPosition(Point2 const &pos) {
211 Point2 new_global_pos = pos + global_start_position_;
212 if (new_global_pos.x() <= 0) {
213 new_global_pos[0] = 0 + kLargeEps;
214 }
215 if (new_global_pos.y() <= 0) {
216 new_global_pos[1] = 0 + kLargeEps;
217 }
218 double max_xy = params_.pWorldMapSize * params_.pResolution;
219 if (new_global_pos.x() >= max_xy) {
220 new_global_pos[0] = max_xy - kLargeEps;
221 }
222 if (new_global_pos.y() >= max_xy) {
223 new_global_pos[1] = max_xy - kLargeEps;
224 }
225
226 local_current_position_ = new_global_pos - global_start_position_;
227 global_current_position_ = new_global_pos;
228 if (params_.pUpdateSensorView == true) {
229 UpdateSensorView();
230 }
231 if (params_.pUpdateRobotMap == true) {
232 UpdateRobotMap();
233 }
234 if (params_.pUpdateExplorationMap == true) {
235 UpdateExplorationMap();
236 }
237 }

322 auto GetExplorationFeatures() {
323 queue_t frontiers;
324 Point2 const &pos = global_current_position_;
325
326 double factor = 4;
327 double pos_x_lim = std::min(pos[0] + factor * time_step_dist_,
328 static_cast<double>(params_.pWorldMapSize));
329 double pos_y_lim = std::min(pos[1] + factor * time_step_dist_,
330 static_cast<double>(params_.pWorldMapSize));
331 for (double pos_x = std::max(pos[0] - factor * time_step_dist_, kLargeEps);
332 pos_x < pos_x_lim; pos_x += 1 * params_.pResolution) {
333 for (double pos_y =
334 std::max(pos[1] - factor * time_step_dist_, kLargeEps);
335 pos_y < pos_y_lim; pos_y += 1 * params_.pResolution) {
336 Point2 qpos{pos_x, pos_y};
337 double dist = (qpos - pos).norm();
338 dist = std::max(dist, time_step_dist_);
339 MapUtils::MapBounds index, offset;
340 MapUtils::ComputeOffsets(params_.pResolution, qpos, params_.pSensorSize,
341 params_.pWorldMapSize, index, offset);
342 double unexplored =
343 exploration_map_
344 .block(index.left + offset.left, index.bottom + offset.bottom,
345 offset.width, offset.height)
346 .sum();
347 double benefit = unexplored;
348 if (unexplored > 0.1 * sensor_area_) {
349 double idf_value =
350 robot_map_
351 .block(index.left + offset.left, index.bottom + offset.bottom,
352 offset.width, offset.height)
353 .count();
354 benefit += 2 * idf_value;
355 }
356 double bcr = benefit / dist;
357
358 if (frontiers.size() < size_t(params_.pNumFrontiers)) {
359 frontiers.push(Frontier(qpos, bcr));
360 } else {
361 auto worst_frontier = frontiers.top();
362 if (worst_frontier.value < bcr) {
363 frontiers.pop();
364 frontiers.push(Frontier(qpos, bcr));
365 }
366 }
367 }
368 }
369 std::vector<double> frontier_features(params_.pNumFrontiers * 2);
370 int count = 0;
371 while (!frontiers.empty()) {
372 auto point = frontiers.top();
373 frontier_features[count++] = point.pt[0] - pos[0];
374 frontier_features[count++] = point.pt[1] - pos[1];
375 frontiers.pop();
376 }
377 return frontier_features;
378 }

380 /* The centroid is computed with orgin of the map, i.e., the lower left corner
381 * of the map. */
382 /* Only the local map is considered for the computation of the centroid. */
383 auto GetVoronoiFeatures() {
384 auto const &pos = global_current_position_;
385 MapUtils::MapBounds index, offset;
386 MapUtils::ComputeOffsets(params_.pResolution, pos, params_.pLocalMapSize,
387 params_.pWorldMapSize, index, offset);
388 auto trimmed_local_map =
389 robot_map_.block(index.left + offset.left, index.bottom + offset.bottom,
390 offset.width, offset.height);
391
392 Point2 map_size(offset.width, offset.height);
393 Point2 map_translation(
394 (index.left + offset.left) * params_.pResolution,
395 (index.bottom + offset.bottom) * params_.pResolution);
396
397 PointVector robot_positions(1);
398 robot_positions[0] = pos - map_translation;
399 Voronoi voronoi(robot_positions, trimmed_local_map, map_size,
400 params_.pResolution, true, 0);
401 auto vcell = voronoi.GetVoronoiCell();
402 /* vcell.centroid += Point2(params_.pLocalMapSize / 2.,
403 * params_.pLocalMapSize / 2.); */
404 return vcell.GetFeatureVector();
405 }
