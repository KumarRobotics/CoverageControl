59
60 h_vals = torch.linspace(1.0, -1.0, self.coverage_maps.shape[-2]+1)
61 h_vals = (h_vals[1:] + h_vals[:-1])/2
62 w_vals = torch.linspace(-1.0, 1.0, self.coverage_maps.shape[-1]+1)
63 w_vals = (w_vals[1:] + w_vals[:-1])/2
64 self.heatmap_x = torch.stack([h_vals]*self.coverage_maps.shape[-1], axis=1)/100
65 self.heatmap_y = torch.stack([w_vals]*self.coverage_maps.shape[-2], axis=0)/100
