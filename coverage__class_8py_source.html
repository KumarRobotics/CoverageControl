22"""
23An example class to use the CoverageControl library to run a coverage algorithm
24"""
25import sys
26import coverage_control as cc # Main library
27from coverage_control import CoverageSystem
28
29# Algorithms available:
30# ClairvoyantCVT
31# CentralizedCVT
32# DecentralizedCVT
33# NearOptimalCVT
34from coverage_control.algorithms import ClairvoyantCVT as CoverageAlgorithm
35
36class RunCoverageAlgorithm:
37 """
38 A class to run the coverage algorithm
39 """
40
41 def __init__(self, params_filename=None):
42 if params_filename is not None:
43 self.params_ = cc.Parameters(params_filename)
44 else:
45 self.params_ = cc.Parameters()
46
47 self.env = CoverageSystem(self.params_)
48 self.controller = CoverageAlgorithm(self.params_, self.params_.pNumRobots, self.env)
49
50 def step(self):
51 """
52 Run one step of the coverage algorithm
53 """
54 self.controller.ComputeActions();
55 actions = self.controller.GetActions()
56 error_flag = self.env.StepActions(actions)
57 return error_flag
58
59 def execute(self):
60 """
61 Run the coverage algorithm
62 """
63 num_steps = 1
64
65 init_cost = self.env.GetObjectiveValue()
66 print(f"Initial Coverage cost: {init_cost:.2e}")
67
68 while num_steps <= self.params_.pEpisodeSteps:
69 if self.step():
70 print(f"Error in step {num_steps}")
71 break
72
73 if self.controller.IsConverged():
74 print(f"Converged in step {num_steps}")
75 break
76
77 num_steps = num_steps + 1
78
79 final_cost = self.env.GetObjectiveValue()
80 print(f"Improvement %: {100 * (init_cost - final_cost)/init_cost:.2f}")
81
82if __name__ == '__main__':
83
84 if len(sys.argv) > 1:
85 cc = RunCoverageAlgorithm(sys.argv[1])
86 else:
87 cc = RunCoverageAlgorithm()
88 cc.execute()
